// Ghost Job Detector — on-page scan (lightweight)
// Runs ONLY on supported job boards based on manifest matches.

const API_BASE_DEFAULT = "https://ghost-job-detector.replit.app";

// --- Utilities ---
function safeHostname(url) {
  try { return new URL(url).hostname; } catch { return ""; }
}

function isSupportedSite(url) {
  const host = safeHostname(url);
  return (
    host.includes("linkedin.com") ||
    host.includes("indeed.com") ||
    host.includes("glassdoor.com") ||
    host.includes("ziprecruiter.com")
  );
}

function setRingState(fab, state) {
  fab.classList.remove("gjdb-ring-neutral", "gjdb-ring-green", "gjdb-ring-yellow", "gjdb-ring-red");
  fab.classList.add(state);
}

function normalizeRisk(risk) {
  const r = String(risk || "").toLowerCase();
  if (r.includes("low") || r.includes("green")) return "green";
  if (r.includes("medium") || r.includes("moderate") || r.includes("yellow")) return "yellow";
  if (r.includes("high") || r.includes("red")) return "red";
  return "neutral";
}

// --- UI ---
function createFab() {
  const fab = document.createElement("div");
  fab.id = "gjdb-fab";
  fab.classList.add("gjdb-ring-neutral");

  const ghost = document.createElement("div");
  ghost.className = "ghost";

  const mouth = document.createElement("div");
  mouth.className = "mouth";
  ghost.appendChild(mouth);

  fab.appendChild(ghost);
  document.documentElement.appendChild(fab);
  return fab;
}

function createTooltip() {
  const tip = document.createElement("div");
  tip.id = "gjdb-tooltip";
  tip.innerHTML = `
    <div class="title">Ghost Job Detector</div>
    <div class="meta">Ready to scan this job posting.</div>
    <div class="pill">Click the ghost to analyze</div>
  `;
  document.documentElement.appendChild(tip);
  return tip;
}

function showTooltip(tip, html) {
  tip.style.display = "block";
  if (html) tip.innerHTML = html;
}

function hideTooltip(tip) {
  tip.style.display = "none";
}

// --- Main ---
(async function init() {
  const pageUrl = window.location.href;

  if (!isSupportedSite(pageUrl)) return; // guardrail

  const fab = createFab();
  const tip = createTooltip();

  // hover behavior
  fab.addEventListener("mouseenter", () => showTooltip(tip));
  fab.addEventListener("mouseleave", () => hideTooltip(tip));

  // load user-configured API (from popup) if present
  function getApiBase() {
    return new Promise((resolve) => {
      chrome.storage.sync.get(["apiBaseUrl"], (result) => {
        resolve(result.apiBaseUrl || API_BASE_DEFAULT);
      });
    });
  }

  let isScanning = false;

  fab.addEventListener("click", async () => {
    if (isScanning) return;
    isScanning = true;

    setRingState(fab, "gjdb-ring-neutral");
    showTooltip(tip, `
      <div class="title">Ghost Job Detector</div>
      <div class="meta">Scanning current page…</div>
      <div class="pill">Working…</div>
    `);

    try {
      const apiBase = await getApiBase();

      // IMPORTANT: adjust endpoint name if yours differs
      const res = await fetch(`${apiBase}/analyze`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ url: pageUrl })
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      const data = await res.json();

      // expected response:
      // { risk: "Low|Moderate|High", score: 0-100, reasons: ["...","..."] }
      const riskState = normalizeRisk(data.risk);
      const ringClass =
        riskState === "green" ? "gjdb-ring-green" :
        riskState === "yellow" ? "gjdb-ring-yellow" :
        riskState === "red" ? "gjdb-ring-red" : "gjdb-ring-neutral";

      setRingState(fab, ringClass);

      const reasons = Array.isArray(data.reasons) ? data.reasons.slice(0, 5) : [];
      const score = (data.score !== undefined && data.score !== null) ? data.score : "—";
      const riskLabel = data.risk || "Unknown";

      showTooltip(tip, `
        <div class="title">Risk Level: ${riskLabel}</div>
        <div class="meta">Score: <b>${score}</b></div>
        <div class="pill">Click again to rescan</div>
        ${reasons.length ? `<ul>${reasons.map(r => `<li>${r}</li>`).join("")}</ul>` : ""}
      `);

    } catch (e) {
      setRingState(fab, "gjdb-ring-neutral");
      showTooltip(tip, `
        <div class="title">Scan failed</div>
        <div class="meta">Couldn’t reach the analyzer.</div>
        <div class="pill">Try again</div>
      `);
    } finally {
      isScanning = false;
    }
  });

})();
